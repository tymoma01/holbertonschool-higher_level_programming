Title: MyList doctests

>>> MyList = __import__('1-my_list').MyList

>>> my_list = MyList()

>>> print(my_list)
[]

>>> my_list.append(4)
>>> print(my_list)
[4]

>>> my_list.append(1)
>>> print(my_list)
[4, 1]
>>> my_list.print_sorted()
[1, 4]

>>> my_list.append(3); my_list.append(2); my_list.append(5)
>>> print(my_list)
[4, 1, 3, 2, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

>>> empty_list = MyList()
>>> empty_list.print_sorted()
[]
>>> print(empty_list)  # original unchanged
[]

>>> a = MyList([3, 1, 2])
>>> result = a.print_sorted()
[1, 2, 3]

>>> b = MyList([5, -2, 0, 5, -2])
>>> print(b)

[-2, -2, 0, 5, 5]
>>> print(b)  # unchanged
[5, -2, 0, 5, -2]
